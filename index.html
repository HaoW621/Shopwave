  const li = document.createElement('li'); li.className = 'spectrum-SideNav-item';

  const btn = document.createElement('button');

  btn.className = 'spectrum-SideNav-link' + (c === currentCat ? ' is-selected' : '');

  btn.innerHTML = `<span>${CAT_LABEL[c]}</span><span class="spectrum-SideNav-count">${cnt}</span>`;

  btn.onclick = () => { currentCat = c; renderNav(); renderProducts(); updateBreadcrumb(); };

  li.appendChild(btn); nav.appendChild(li);

});


function renderNav() {

  nav.querySelectorAll('.spectrum-SideNav-link').forEach((b, i) => b.classList.toggle('is-selected', CATS[i] === currentCat));

}

function updateBreadcrumb() {

  document.getElementById('breadcrumbCat').textContent = CAT_LABEL[currentCat] || currentCat;

}

function filterProducts() {

  searchQ = document.getElementById('searchInput').value.trim().toLowerCase();

  renderProducts();

}

function renderProducts() {

  const grid = document.getElementById('productGrid');

  const title = document.getElementById('sectionTitle');

  const cnt = document.getElementById('resultCount');

  let filtered = PRODUCTS.filter(p =>

    (currentCat === 'すべて' || p.cat === currentCat) &&

    (!searchQ || p.name.toLowerCase().includes(searchQ) || p.cat.includes(searchQ) || p.desc.includes(searchQ))

  );

  title.textContent = CAT_LABEL[currentCat] || currentCat;

  cnt.textContent = `${filtered.length}件`;

  grid.innerHTML = '';

  if (!filtered.length) {

    grid.innerHTML = `<div class="spectrum-EmptyState" style="grid-column:1/-1">

      <div class="spectrum-EmptyState-icon">🔍</div>

      <div class="spectrum-EmptyState-heading">該当する商品が見つかりません</div>

      <p style="font-size:13px">検索ワードやカテゴリーを変更してみてください。</p>

    </div>`; return;

  }

  filtered.forEach(p => {

    const el = document.createElement('div'); el.className = 'spectrum-Card';

    el.innerHTML = `

      <div class="spectrum-Card-preview">

        ${p.emoji}

        ${p.isNew ? '<span class="spectrum-Tag spectrum-Tag--blue spectrum-Card-badge">NEW</span>' : ''}

      </div>

      <div class="spectrum-Card-body">

        <div class="spectrum-Card-subtitle">${p.cat}</div>

        <div class="spectrum-Card-title">${p.name}</div>

        <div class="spectrum-Card-desc">${p.desc}</div>

        <div class="spectrum-Card-footer">

          <div class="spectrum-Card-price">¥${p.price.toLocaleString()} <small>税込</small></div>

          <button class="spectrum-Button spectrum-Button--cta" style="height:28px;padding:0 12px;font-size:12px" onclick="addToCart(${p.id})">+ 追加</button>

        </div>

      </div>`;

    grid.appendChild(el);

  });

}


function addToCart(id) {

  cart[id] = (cart[id] || 0) + 1;

  updateCartBadge();

  const p = PRODUCTS.find(x => x.id === id);

  showToast(`✅ ${p.name} をカートに追加しました`);

}

function removeFromCart(id) { delete cart[id]; updateCartBadge(); renderCart(); }

function changeQty(id, d) {

  cart[id] = (cart[id] || 0) + d;

  if (cart[id] <= 0) delete cart[id];

  updateCartBadge(); renderCart();

}

function updateCartBadge() {

  document.getElementById('cartBadge').textContent = Object.values(cart).reduce((a,b)=>a+b,0);

}

function renderCart() {

  const body = document.getElementById('cartBody'), footer = document.getElementById('cartFooter');

  const keys = Object.keys(cart);

  if (!keys.length) {

    body.innerHTML = `<div class="spectrum-Cart-empty"><div class="spectrum-Cart-empty-icon">🛒</div><p>カートに商品がありません</p></div>`;

    footer.style.display = 'none'; return;

  }

  footer.style.display = '';

  let total = 0;

  body.innerHTML = keys.map(id => {

    const p = PRODUCTS.find(x => x.id == id), sub = p.price * cart[id]; total += sub;

    return `<div class="spectrum-CartItem">

      <div class="spectrum-CartItem-icon">${p.emoji}</div>

      <div class="spectrum-CartItem-info">

        <div class="spectrum-CartItem-name">${p.name}</div>

        <div class="spectrum-CartItem-price">¥${p.price.toLocaleString()} × ${cart[id]} = ¥${sub.toLocaleString()}</div>

        <div class="spectrum-Stepper">

          <button class="spectrum-Stepper-btn" onclick="changeQty(${id},-1)">−</button>

          <span class="spectrum-Stepper-value">${cart[id]}</span>

          <button class="spectrum-Stepper-btn" onclick="changeQty(${id},1)">+</button>

          <button class="spectrum-ActionButton" onclick="removeFromCart(${id})" style="margin-left:6px;font-size:12px" title="削除">🗑</button>

        </div>

      </div>

    </div>`;

  }).join('');

  document.getElementById('totalPrice').textContent = '¥' + total.toLocaleString();

}

function toggleCart() {

  const open = document.getElementById('drawer').classList.toggle('is-open');

  document.getElementById('overlay').classList.toggle('is-open', open);

  if (open) renderCart();

}

function checkout() {

  cart = {}; updateCartBadge(); toggleCart();

  showToast('✅ ご注文ありがとうございます！');

}

function showToast(msg) {

  const t = document.getElementById('toast'); t.textContent = msg;

  t.classList.add('is-open');

  setTimeout(() => t.classList.remove('is-open'), 2800);

}
